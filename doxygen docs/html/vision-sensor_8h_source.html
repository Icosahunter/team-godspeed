56  double CalculateDistance(double obj_w, double obj_h, double obj_x, double obj_w_actual, double obj_h_actual)
57  {
58  double orth_dist_v = ((obj_h_actual/obj_h)*ScreenHeight/2)/std::tan(VerticalFOV);
59  double orth_dist_h = ((obj_w_actual/obj_w)*ScreenWidth/2)/std::tan(HorizontalFOV);
60  double orth_dist = std::min(orth_dist_v, orth_dist_h);
61  double ratio = std::min(obj_h_actual/obj_h, obj_w_actual/obj_w);
62  double leg1 = ratio*obj_x + ratio*obj_w/2 - ratio*ScreenWidth/2;
63  double leg2 = orth_dist;
64  return std::sqrt(std::pow(leg1, 2) + std::pow(leg2, 2));
65  }
